# Замена звуковой дорожки
ffmpeg -i input.mp4 -i input.mp3 -codec copy -shortest output.mp4
ffmpeg -i input.mp4 -i input.aac -codec copy -bsf:a aac_adtstoasc -shortest output.mp4

# Нарезка видео на ts-файлы для кодеков h264/hevc
ffmpeg -i input.mp4 -c copy -bsf:v h264_mp4toannexb -hls_time 10 -hls_key_info_file file.keyinfo -hls_list_size 0 output.m3u8
ffmpeg -i input.mp4 -c copy -bsf:v hevc_mp4toannexb -hls_time 10 -hls_key_info_file file.keyinfo -hls_list_size 0 output.m3u8

# Удаление аудиодорожки из видео
ffmpeg -i WithSound.mp4 -vcodec copy -an noSound.mp4

# Извлечение аудиодорожки из видео
ffmpeg -i input_file_with_mp3_audio.mp4 -vn -ar 44100 -ac 2 -ab 192K -f mp3 audiotrack.mp3
ffmpeg -i input_file_with_aac_audio.mp4 -c:a copy audiotrack.aac

# Изменение битрейта видео
ffmpeg -i input.mp4 -c:v libx264 -b:v 16000K output.mp4
